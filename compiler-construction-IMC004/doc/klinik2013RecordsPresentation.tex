\begin{frame}[fragile]{But What \emph{is} a Record?}

\onslide<+->

\begin{itemize}
  \item How is it stored in memory?
  \item How does the code for record projection look like?
\end{itemize}

\onslide<+->

The code

\begin{verbatim}
a getX(b b) { return b.x; }
\end{verbatim}

must work for:

\begin{verbatim}
getX({x=10})
getX({y=True,x=False})
\end{verbatim}

and any other combination of fields.

\end{frame}
